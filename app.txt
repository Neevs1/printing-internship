from flask import Flask, render_template, request, jsonify
import pandas as pd
import joblib
import json
import os

app = Flask(__name__)

# File Paths
BASE_DIR = os.path.dirname(os.path.abspath(__file__))
MODEL_PATH = os.path.join(BASE_DIR, r'C:\Users\ARJUN\OneDrive\Desktop\printing-internship\ink_model_pipeline.pkl')
EXCEL_PATH = os.path.join(BASE_DIR, 'print_logs.xlsx')
CONFIG_PATH = os.path.join(BASE_DIR, 'press_settings.json')

# Helper: Load Press Config
def get_config():
    if os.path.exists(CONFIG_PATH):
        with open(CONFIG_PATH, 'r') as f:
            return json.load(f)
    return {"density_relation": 0.5, "zero_setting": 0.0, "target_de": 2.5}

@app.route('/')
def index():
    return render_template('index.html', config=get_config())

@app.route('/settings', methods=['GET', 'POST'])
def settings():
    if request.method == 'POST':
        config = {
            "density_relation": float(request.form['density_relation']),
            "zero_setting": float(request.form['zero_setting']),
            "target_de": float(request.form['target_de'])
        }
        with open(CONFIG_PATH, 'w') as f:
            json.dump(config, f)
        return render_template('settings.html', config=config, msg="Settings Saved Locally!")
    return render_template('settings.html', config=get_config())

@app.route('/predict', methods=['POST'])
def predict():
    try:
        data = request.json
        config = get_config()
        model = joblib.load(MODEL_PATH)

        # 1. Feature Engineering (Match notebook requirements)
        de_before = float(data['de_before'])
        target_de = float(config['target_de'])
        
        input_df = pd.DataFrame([{
            'Press_ID': 1,
            'Job_Number': str(data['job_number']),
            'Paper_type': str(data['paper_type']),
            'Zone_number': float(data['zone_number']),
            'Color': str(data['color']),
            'Ink_key_zero_setting': float(config['zero_setting']),
            'Delta_E_before': de_before,
            'initial_density': float(data.get('init_dens', 0) or 0),
            'final_density': float(data.get('init_dens', 0) or 0) + float(config['density_relation']),
            'initial_ink_key_setting': float(data['init_key']),
            'DeltaE_reduction': de_before - target_de
        }])

        # 2. ML Prediction
        prediction = model.predict(input_df)
        final_key = round(float(prediction[0]), 2)

        # 3. Initial Log to Excel
        log_df = input_df.copy()
        log_df['final_ink_key_setting'] = final_key # Default to prediction until verified
        
        if not os.path.exists(EXCEL_PATH):
            log_df.to_excel(EXCEL_PATH, index=False)
        else:
            existing = pd.read_excel(EXCEL_PATH)
            pd.concat([existing, log_df], ignore_index=True).to_excel(EXCEL_PATH, index=False)

        return jsonify({"status": "success", "final_key": final_key, "final_de": target_de})
    except Exception as e:
        return jsonify({"status": "error", "message": str(e)})

@app.route('/save_actual', methods=['POST'])
def save_actual():
    try:
        actual_val = float(request.json['actual_key'])
        df = pd.read_excel(EXCEL_PATH)
        # Update the target column of the very last entry with the user-verified value
        df.iloc[-1, df.columns.get_loc('final_ink_key_setting')] = actual_val
        df.to_excel(EXCEL_PATH, index=False)
        return jsonify({"status": "success"})
    except Exception as e:
        return jsonify({"status": "error", "message": str(e)})

@app.route('/retrain', methods=['POST'])
def retrain():
    try:
        # Load logs and drop any rows missing a target
        data = pd.read_excel(EXCEL_PATH).dropna(subset=['final_ink_key_setting'])
        X = data.drop(['final_ink_key_setting', 'Delta_E_after'], axis=1, errors='ignore')
        y = data['final_ink_key_setting']
        
        # Load current brain and re-fit with new data
        model_pipe = joblib.load(MODEL_PATH)
        model_pipe.fit(X, y)
        joblib.dump(model_pipe, MODEL_PATH)
        
        return jsonify({"status": "success", "message": f"Model learned from {len(data)} jobs!"})
    except Exception as e:
        return jsonify({"status": "error", "message": str(e)})

if __name__ == '__main__':
    app.run(debug=True, port=5000)
