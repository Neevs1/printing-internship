from flask import Flask, render_template, request, jsonify
import pandas as pd
import joblib
import os

app = Flask(__name__)

# Load the model you saved from your notebook
# Ensure the .pkl file is in the same folder as this script
model_path = os.path.join(os.path.dirname(__file__), 'ink_model_pipeline.pkl')

try:
    model = joblib.load(model_path)
    print("Model loaded successfully!")
except:
    print("Error: ink_model_pipeline.pkl not found. Please export it from your notebook first.")

@app.route('/')
def index():
    return render_template('index.html')

@app.route('/predict', methods=['POST'])
def predict():
    try:
        data = request.json
        
        # We must create a DataFrame with the EXACT names from your notebook
        input_df = pd.DataFrame([{
            'Press_ID': int(data['Press_ID']),
            'Job_Number': str(data['Job_Number']),
            'Paper_type': str(data['Paper_type']),
            'Zone_number': float(data['Zone_number']),
            'Color': str(data['Color']),
            'Ink_key_zero_setting': float(data['Ink_key_zero_setting']),
            'Delta_E_before': float(data['Delta_E_before']),
            'initial_density': float(data['initial_density']),
            'final_density': float(data['final_density']),
            'initial_ink_key_setting': float(data['initial_ink_key_setting']),
            'final_ink_key_setting': float(data['final_ink_key_setting'])
        }])

        # Run the XGBoost Prediction
        prediction = model.predict(input_df)
        
        return jsonify({
            "status": "success",
            "prediction": round(float(prediction[0]), 3)
        })

    except Exception as e:
        return jsonify({"status": "error", "message": str(e)})

if __name__ == '__main__':
    app.run(debug=True)
